% Define relation operators
\newcommand{\definerelation_operators}{
    \defineop{\lq}{\leq}        % Less than or equal to
    \defineop{\gq}{\geq}        % Greater than or equal to
    \defineop{\lqs}{\leqslant}  % Less than or equal to (slant)
    \defineop{\gqs}{\geqslant}  % Greater than or equal to (slant)
    \defineop{\prc}{\prec}      % Precedes
    \defineop{\suc}{\succ}      % Succeeds
    \defineop{\prcq}{\preceq}   % Precedes or equal
    \defineop{\suceq}{\succeq}  % Succeeds or equal
    \defineop{\sbst}{\subset}   % Subset
    \defineop{\spst}{\supset}   % Superset
    \defineop{\sbsteq}{\subseteq} % Subset or equal
    \defineop{\spsteq}{\supseteq} % Superset or equal
    \defineop{\sqsbst}{\sqsubset} % Square subset
    \defineop{\sqspst}{\sqsupset} % Square superset
    \defineop{\sqsbsteq}{\sqsubseteq} % Square subset or equal
    \defineop{\sqspsteq}{\sqsupseteq} % Square superset or equal
    \defineop{\deq}{\doteq}     % Dotted equals
    \newcommand{\eqv}{\equiv}   % Equivalent
    \newcommand{\aprx}{\approx} % Approximately
    \newcommand{\cng}{\cong}    % Congruent
    \newcommand{\smeq}{\simeq}  % Similar or equal
    \newcommand{\sm}{\sim}      % Similar
    \newcommand{\prop}{\propto} % Proportional
}
